import pandas as pd

# 讀取檔案
df = pd.read_csv('活頁簿1.csv')

# 設定目標欄位
target_col = '重現精度(mm)'

# 分割資料
# 1. 取出重現精度
df_repeatability = df[[target_col]]

# 2. 取出其餘資料 (刪除重現精度欄位)
df_others = df.drop(columns=[target_col])

# 儲存結果
df_repeatability.to_csv('repeatability_part.csv', index=False, encoding='utf-8-sig')
df_others.to_csv('others_part.csv', index=False, encoding='utf-8-sig')

# 顯示前幾筆資料確認
print("=== 重現精度 ===")
print(df_repeatability.head())
print("\n=== 其餘資料 ===")
print(df_others.head())
